### Method 3: Browser Console Script (Inline)

Run this script in the browser console (F12) on https://www.dwyeromega.com/en-us/ to extract color values:

```javascript
// Color Extraction Script for DwyerOmega
(function () {
  const colors = {};

  // Helper function to get computed color
  function getColor(selector, property = 'color') {
    const el = document.querySelector(selector);
    if (!el) return null;
    const style = window.getComputedStyle(el);
    return style[property];
  }

  // Helper function to convert RGB to Hex
  function rgbToHex(rgb) {
    if (!rgb || rgb === 'transparent' || rgb === 'rgba(0, 0, 0, 0)') return null;
    const match = rgb.match(/\d+/g);
    if (!match || match.length < 3) return null;
    const r = parseInt(match[0]).toString(16).padStart(2, '0');
    const g = parseInt(match[1]).toString(16).padStart(2, '0');
    const b = parseInt(match[2]).toString(16).padStart(2, '0');
    return `#${r}${g}${b}`.toUpperCase();
  }

  // Extract navigation link color (primary blue)
  const navLink =
    document.querySelector('a[href*="/en-us/"]') ||
    document.querySelector('.header__nav a') ||
    document.querySelector('nav a');
  if (navLink) {
    const color = window.getComputedStyle(navLink).color;
    colors.primaryBlue = {
      rgb: color,
      hex: rgbToHex(color),
    };
  }

  // Extract search bar background
  const searchBar =
    document.querySelector('input[type="search"]') ||
    document.querySelector('input[placeholder*="Search"]');
  if (searchBar) {
    const bg = window.getComputedStyle(searchBar).backgroundColor;
    colors.searchBackground = {
      rgb: bg,
      hex: rgbToHex(bg),
    };
  }

  // Extract top bar background
  const topBar =
    document.querySelector('.header__top') || document.querySelector('[class*="top-bar"]');
  if (topBar) {
    const bg = window.getComputedStyle(topBar).backgroundColor;
    colors.topBarBackground = {
      rgb: bg,
      hex: rgbToHex(bg),
    };
  }

  // Extract body text color
  const body = document.body;
  if (body) {
    const color = window.getComputedStyle(body).color;
    colors.bodyText = {
      rgb: color,
      hex: rgbToHex(color),
    };
  }

  // Extract heading color
  const heading = document.querySelector('h1') || document.querySelector('h2');
  if (heading) {
    const color = window.getComputedStyle(heading).color;
    colors.headingText = {
      rgb: color,
      hex: rgbToHex(color),
    };
  }

  // Extract font families
  if (body) {
    colors.bodyFont = window.getComputedStyle(body).fontFamily;
  }
  if (heading) {
    colors.headingFont = window.getComputedStyle(heading).fontFamily;
  }

  // Output results
  console.log('=== DwyerOmega Color Extraction ===');
  console.log(JSON.stringify(colors, null, 2));
  console.log('=== Copy the above JSON and update DWYEROMEGA_BRAND_FINDINGS.md ===');

  return colors;
})();
```
